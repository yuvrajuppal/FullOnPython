```python
from fastapi import FastAPI
app = FastAPI()
@app.get('/')
def hello():
	return {"message":"hello world"}
```

```python
uvicorn main:app --reload --port 3000
```


```python
@app.get('/profile/{username}')
def profile(username):
return {
"username":username,
}

  

@app.get('/profile/{username}/{comments}')
def profile(username,comments):
return {
"username":username,
"comment":comments
}


# path with type 
@app.get('/product/{id}')
def productid(id:int):
return {
"productid":id,
}
```
