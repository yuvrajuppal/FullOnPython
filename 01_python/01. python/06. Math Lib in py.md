```python
import math

# Using math.pi (π ≈ 3.14159...)
print("pi =", math.pi)  # 3.141592653589793

# Using math.e (e ≈ 2.71828...)
print("e =", math.e)    # 2.718281828459045

# Square root using math.sqrt()
print("sqrt(25) =", math.sqrt(25))        # 5.0
print("sqrt(2) =", math.sqrt(2))          # 1.4142135623730951

# math.ceil(x) → smallest integer >= x
print("ceil(1.4) =", math.ceil(1.4))      # 2
print("ceil(-1.4) =", math.ceil(-1.4))    # -1

# math.floor(x) → largest integer <= x
print("floor(1.4) =", math.floor(1.4))    # 1
print("floor(-1.4) =", math.floor(-1.4))  # -2

```
