```python
import random

# 1. Rolling a 6-sided die
dice_roll = random.randint(1, 6)
print(f"You rolled a: {dice_roll}")

# 2. Choosing a random encounter
enemies = ['Goblin', 'Dragon', 'Slime', 'Skeleton']
spawn = random.choice(enemies)
print(f"A wild {spawn} appeared!")

# 3. Shuffling a deck of cards
deck = ['Ace', 'King', 'Queen', 'Jack', '10']
random.shuffle(deck)
print(f"The shuffled deck is: {deck}")

# 4. Picking 2 random lucky winners (no repeats)
contestants = ['Alice', 'Bob', 'Charlie', 'David', 'Eve']
winners = random.sample(contestants, 2)
print(f"The winners are: {winners}")
```
